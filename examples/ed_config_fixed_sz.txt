# Example ED configuration with Fixed Sz mode
# For a 12-site Heisenberg chain at half-filling (Sz=0)

method=lanczos

# System parameters

spin_length=0.5

# Fixed Sz mode
use_fixed_sz=true
n_up=6          # Half-filling: Sz = 6 - 12/2 = 0

# Solver parameters
num_eigenvalues=5
max_iterations=10000
tolerance=1e-10

# Workflow
run_standard=true
output_dir=output_fixed_sz
